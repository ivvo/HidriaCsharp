'------------------------------------------------------------MOVE AWAY
*robot_moveAway:
    '
    'if robot is INSIDE station, move to OUTSIDE
    'if not, SKIP
    IF in_station=1 THEN
        '
        'call the function to set the current power state (LOW or HIGH)
        GOSUB *robot_setPowerMode
        '
        'set CONTPLS for continuous movements
        CALL *robot_setCONTPLS ("vertical")
        '
        'write the current position to a point, for movement relative to current position
        pAboveStation = WHRXY
        '
        'move the robot to OUTSIDE of the current station
{0}
        ELSE
            'output error
            SET O_PROGRAM_ERROR
            HOLD
            '
        ENDIF
        '
        'robot is OUTSIDE station
        in_station=0
        '
    ENDIF
    '
    'robot is OUTSIDE of station so set fullWorkingSpeed
    GOSUB *robot_fullWorkingSpeed
    '
RETURN
