Function {0}_setPowerMode ' choose between LOW(safe) and HIGH(full power) power modes
	Robot {2}
	
	If MemSw({1}_LOCAL_RESET_FLAG) = On Or Sw({1}_I_STEP) = On Or MemSw({1}_LOCAL_HOMING_FLAG) = On OR 1=1 Then
		'if the machine is in STEP mode, or robot is performing RESET or HOMING, set power mode to LOW
		If Power <> Low Then
			WaitPos
			Wait 0.1
			Power Low
		EndIf
				
	Else
		'if the machine is in AUTO mode set power mode to HIGH
		If Power <> High Then
			WaitPos
			Power Low	'Change to High
		EndIf
	EndIf

	Call robot_fullWorkingSpeed

Fend
